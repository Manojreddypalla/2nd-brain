### âœ… 1. Print Array Elements

**Question:** Input N elements and print them.

**Logic:** Simple traversal.

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[n];

    for (int i = 0; i < n; i++) cin >> arr[i];

    cout << "Array elements: ";
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
}

```

### âœ… 2. Find Maximum and Minimum

**Question:** Find the largest and smallest element in an array.

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[n];

    for (int i = 0; i < n; i++) cin >> arr[i];

    int max = arr[0], min = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > max) max = arr[i];
        if (arr[i] < min) min = arr[i];
    }

    cout << "Max: " << max << ", Min: " << min;
}

```

ðŸ’¡ _Concept:_ Comparison inside a loop.

### âœ… 3. Find Sum and Average

**Question:** Find sum and average of all elements.

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[n];

    for (int i = 0; i < n; i++) cin >> arr[i];

    int sum = 0;
    for (int i = 0; i < n; i++)
        sum += arr[i];

    double avg = (double)sum / n;
    cout << "Sum = " << sum << ", Average = " << avg;
}

```

ðŸ’¡ _Concept:_ Accumulation (running total).

### âœ… 4. Count Even and Odd Numbers

**Question:** Count how many numbers are even and how many are odd.

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[n];

    for (int i = 0; i < n; i++) cin >> arr[i];

    int even = 0, odd = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] % 2 == 0)
            even++;
        else
            odd++;
    }

    cout << "Even = " << even << ", Odd = " << odd;
}

```

ðŸ’¡ _Concept:_ Counting based on a condition.

### âœ… 5. Reverse the Array

**Question:** Reverse an array **in-place** (no extra array).

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[n];

    for (int i = 0; i < n; i++) cin >> arr[i];

    int start = 0, end = n - 1;
    while (start < end) {
        int temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;
        start++;
        end--;
    }

    cout << "Reversed array: ";
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
}

```

ðŸ’¡ _Concept:_ Swapping using two-pointer logic.
