# âœ… **1. Maximum Sum Subarray of Size K (Fixed Window)**

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    vector<int> arr = {2, 5, 1, 8, 2, 9};
    int k = 3;

    int n = arr.size();
    int sum = 0, maxSum = 0;

    // First window
    for (int i = 0; i < k; i++) sum += arr[i];
    maxSum = sum;

    // Sliding window
    for (int i = k; i < n; i++) {
        sum += arr[i] - arr[i - k];
        maxSum = max(maxSum, sum);
    }

    cout << "Max sum = " << maxSum;
}
```
