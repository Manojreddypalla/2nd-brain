so the pattern is copy the array to a set and compare the size and if not equal there are duplicates

```cpp
class Solution {

public:

    bool hasDuplicate(vector<int>& nums) {

        unordered_set<int> s(nums.begin(),nums.end());

        if(nums.size()!=s.size())

        {

            return true;

        }

        return false;

    }

};
```
