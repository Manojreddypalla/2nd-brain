class Solution {

public:

  

     vector<vector<int>> ans;

    void solve(TreeNode * root,int level)

    {

        if(root==nullptr)

        {

            return;

        }

        if(ans.size()==level)

        {

            ans.push_back({});

        }

        ans[level].push_back(root->val);

        solve(root->left,level+1);

        solve(root->right,level+1);

  

  

    }

    vector<vector<int>> levelOrder(TreeNode* root) {

        solve(root,0);

        return ans;

    }

};
